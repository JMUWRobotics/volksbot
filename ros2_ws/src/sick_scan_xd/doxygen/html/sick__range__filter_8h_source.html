68  /*
69  * enum RangeFilterResultHandling configures the range filter handling
70  * Note: Range filter applies only to Pointcloud messages, not to LaserScan messages.
71  */
72  typedef enum RangeFilterResultHandlingEnum
73  {
74  RANGE_FILTER_DEACTIVATED = 0, // do not apply range filter
75  RANGE_FILTER_DROP = 1, // drop point, if range is not within [range_min, range_max]
76  RANGE_FILTER_TO_ZERO = 2, // set range = 0, if range is not within [range_min, range_max]
77  RANGE_FILTER_TO_RANGE_MAX = 3, // set range = range_max, if range is not within [range_min, range_max]
78  RANGE_FILTER_TO_FLT_MAX = 4, // set range = FLT_MAX, if range is not within [range_min, range_max]
79  RANGE_FILTER_TO_NAN = 5 // set range = NAN, if range is not within [range_min, range_max]
80  } RangeFilterResultHandling;

82  /*
83  * class SickRangeFilter filters a pointcloud by range.
84  */
85  class SickRangeFilter
86  {
87  public:
88 
89  /*
90  * Initializing constructor
91  * @param[in] range_min min range in meter, default: 0
92  * @param[in] range_max max range in meter, default: FLT_MAX
93  * @param[in] settings filter configuration, default: RANGE_FILTER_DEACTIVATED (do not apply range filter)
94  */
95  SickRangeFilter(float range_min = 0, float range_max = FLT_MAX, RangeFilterResultHandling settings = RangeFilterResultHandling::RANGE_FILTER_DEACTIVATED)
96  : m_range_min(range_min), m_range_max(range_max), m_settings(settings)
97  {
98  }
99 
100  /*
101  * Apply an optional range filter.
102  * Note: Range filter applies only to Pointcloud messages, not to LaserScan messages.
103  * @param[in+out] range range in meter (input: range from scan message, output: range after filter)
104  * @param[out] range_modified true, if range has been overwritten, otherwise false
105  * @return false if point dropped, otherwise true (i.e. append a point to the pointcloud, if apply returns true)
106  */
107  bool apply(float& range, bool& range_modified) const
108  {
109  bool ret_val = true;
110  range_modified = false;
111  if ((m_settings != RANGE_FILTER_DEACTIVATED) && (range < m_range_min || range > m_range_max)) // range not in [range_min, range_max], apply filter
112  {
113  switch(m_settings)
114  {
115  case RANGE_FILTER_DEACTIVATED: // do not apply range filter
116  break;
117  case RANGE_FILTER_DROP: // drop point, if range is not within [range_min, range_max]
118  ret_val = false;
119  break;
120  case RANGE_FILTER_TO_ZERO: // set range = 0, if range is not within [range_min, range_max]
121  range = 0;
122  range_modified = true;
123  break;
124  case RANGE_FILTER_TO_RANGE_MAX: // set range = range_max, if range is not within [range_min, range_max]
125  range = m_range_max;
126  range_modified = true;
127  break;
128  case RANGE_FILTER_TO_FLT_MAX: // set range = FLT_MAX, if range is not within [range_min, range_max]
129  range = FLT_MAX;
130  range_modified = true;
131  break;
132  case RANGE_FILTER_TO_NAN: // set range = NAN, if range is not within [range_min, range_max]
133  range = std::nanf("");
134  range_modified = true;
135  break;
136  default:
137  ROS_ERROR_STREAM("## ERROR SickRangeFilter::apply(): invalid setting " << m_settings << ", please check parameter \"range_filter_handling\" in the configuration and/or launch-file.");
138  break;
139  }
140  }
141  return ret_val;
142  }

144  /*
145  * Overwrites x,y,z of a lidar_point if range has been set to 0, max, FLT_MAX or NaN by apply
146  * @return true if x,y,z have been set, false otherwise
147  */
148  bool applyXYZ(float& x, float& y, float& z, float azimuth, float elevation)
149  {
150  bool ret_val = false;
151  switch(m_settings)
152  {
153  case RANGE_FILTER_TO_ZERO:
154  x = y = z = 0.0f;
155  ret_val = true;
156  break;
157  case RANGE_FILTER_TO_RANGE_MAX:
158  x = m_range_max * cos(elevation) * cos(azimuth);
159  y = m_range_max * cos(elevation) * sin(azimuth);
160  z = m_range_max * sin(elevation);
161  ret_val = true;
162  break;
163  case RANGE_FILTER_TO_FLT_MAX:
164  x = y = z = FLT_MAX;
165  ret_val = true;
166  break;
167  case RANGE_FILTER_TO_NAN:
168  x = y = z = std::nanf("");;
169  ret_val = true;
170  break;
171  default:
172  break;
173  }
174  return ret_val;
175  }

177  /*
178  * Copy and resize a pointcloud after applying an optional range filter.
179  * @param[in] rangeNum number of points after range filtering
180  * @param[in+out] pointcloud pointcloud with resized data fields
181  */
182  void resizePointCloud(size_t rangeNum, ros_sensor_msgs::PointCloud2& cloud)
183  {
184  ros_sensor_msgs::PointCloud2 resized_pointcloud;
185  resized_pointcloud.header = cloud.header;
186  resized_pointcloud.height = cloud.height;
187  resized_pointcloud.width = rangeNum;
188  resized_pointcloud.is_bigendian = cloud.is_bigendian;
189  resized_pointcloud.is_dense = cloud.is_dense;
190  resized_pointcloud.point_step = cloud.point_step;
191  resized_pointcloud.fields = cloud.fields;
192  resized_pointcloud.row_step = resized_pointcloud.point_step * resized_pointcloud.width;
193  resized_pointcloud.data.resize(resized_pointcloud.row_step * resized_pointcloud.height, 0);
194  for(size_t n = 0; n < resized_pointcloud.height; n++) // copy pointcloud data to flat array
195  {
196  memcpy(&resized_pointcloud.data[n * resized_pointcloud.row_step], &cloud.data[n * cloud.row_step], resized_pointcloud.row_step);
197  }
198  cloud = resized_pointcloud;
199  }
