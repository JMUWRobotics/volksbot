294 template <typename T> bool rosGetParam(rosNodePtr nh, const std::string& param_name, T& param_value)
295 {
296  try
297  {
298  bool bRet = nh->get_parameter(param_name, param_value);
299  ROS_DEBUG_STREAM("rosGetParam(" << param_name << "): " << paramToString(param_value) << ", " << typeid(param_value).name());
300  return bRet;
301  }
302  catch(const std::exception& exc)
303  {
304  ROS_WARN_STREAM("## WARNING rosGetParam(" << param_name << ", " << paramToString(param_value) << ", " << typeid(param_value).name() << ") failed, " << typeid(exc).name() << ", exception " << exc.what());
305  }
306  try
307  {
308  std::string str_value;
309  bool bRet = nh->get_parameter(param_name, str_value);
310  if (std::is_arithmetic<T>::value)
311  {
312  rosConvParam(str_value, param_value);
313  ROS_INFO_STREAM("rosGetParam(" << param_name << "): converted to " << param_value);
314  return bRet;
315  }
316  else
317  {
318  ROS_WARN_STREAM("## WARNING rosGetParam(" << param_name << ", " << paramToString(param_value) << ") failed.");
319  }
320  }
321  catch(const std::exception& exc)
322  {
323  ROS_WARN_STREAM("## WARNING rosGetParam(" << param_name << ", " << paramToString(param_value) << ", " << typeid(param_value).name() << ") failed, " << typeid(exc).name() << ", exception " << exc.what());
324  }
325  return false;
326 }

351 class QoSConverter
352 {
353 public:
354  int convert(const rosQoS& qos) const
355  {
356  for (std::map<int,rosQoS>::const_iterator qos_iter = m_qos_map.cbegin(); qos_iter != m_qos_map.cend(); qos_iter++)
357  if (qos_iter->second == qos)
358  return qos_iter->first;
359  return 0;
360  }
361  rosQoS convert(const int& qos) const
362  {
363  for (std::map<int,rosQoS>::const_iterator qos_iter = m_qos_map.cbegin(); qos_iter != m_qos_map.cend(); qos_iter++)
364  if (qos_iter->first == qos)
365  return qos_iter->second;
366  return rclcpp::SystemDefaultsQoS();
367  }
368 protected:
369  std::map<int,rosQoS> m_qos_map = { {0, rclcpp::SystemDefaultsQoS()}, {1, rclcpp::ParameterEventsQoS()}, {2, rclcpp::ServicesQoS()}, {3, rclcpp::ParametersQoS()}, {4, rclcpp::SensorDataQoS()} };
370 };
