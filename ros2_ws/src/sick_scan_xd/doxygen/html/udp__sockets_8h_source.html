196  size_t Receive(std::vector<uint8_t>& msg_payload)
197  {
198  int64_t bytes_received = 0;
199  while (m_running && bytes_received <= 0)
200  {
201  bytes_received = recv(m_udp_socket, (char*)msg_payload.data(), (int)msg_payload.size(), m_recv_flags);
202  if (m_recv_blocking && bytes_received < 0)
203  return 0; // socket error
204  }
205  if (!m_running || bytes_received < 0)
206  return 0; // socket error
207  return (size_t)bytes_received;
208  }

211  size_t Receive(std::vector<uint8_t>& msg_payload, double timeout, const std::vector<uint8_t>& udp_msg_start_seq)
212  {
213  chrono_system_time start_timestamp = chrono_system_clock::now();
214  size_t headerlength = udp_msg_start_seq.size() + sizeof(uint32_t); // 8 byte header: 0x02020202 + Payloadlength
215  size_t bytes_received = 0;
216  size_t bytes_to_receive = msg_payload.size();
217  // Receive \x02\x02\x02\x02 | 4Bytes payloadlength incl. CRC | Payload | CRC32
218  while (m_running && bytes_received < bytes_to_receive && (timeout < 0 || sick_scansegment_xd::Seconds(start_timestamp, chrono_system_clock::now()) < timeout))
219  {
220  int64_t chunk_bytes_received = recv(m_udp_socket, (char*)msg_payload.data() + bytes_received, (int)msg_payload.size() - bytes_received, m_recv_flags);
221  if (chunk_bytes_received <= 0)
222  {
223  std::this_thread::sleep_for(std::chrono::milliseconds(1));
224  continue;
225  }
226  // std::cout << "UdpSenderSocketImpl::Receive(): chunk of " << std::dec << chunk_bytes_received << " bytes received: " << std::endl;
227  // for(int n = 0; n < chunk_bytes_received; n++)
228  // std::cout << std::setfill('0') << std::setw(2) << std::hex << (int)(msg_payload.data()[bytes_received + n] & 0xFF);
229  // std::cout << std::endl;
230  if (bytes_received == 0 && chunk_bytes_received > (int64_t)headerlength && std::equal(msg_payload.begin(), msg_payload.begin() + udp_msg_start_seq.size(), udp_msg_start_seq.begin())) // start of new msgpack
231  {
232  // Start of new message: restart timeout
233  start_timestamp = chrono_system_clock::now();
234  // Decode 8 byte header: 0x02020202 + Payloadlength
235  size_t Payloadlength= Convert4Byte(msg_payload.data() + udp_msg_start_seq.size());
236  bytes_to_receive = Payloadlength + headerlength + sizeof(uint32_t); // 8 byte header + payload + 4 byte CRC
237  if(bytes_to_receive > msg_payload.size())
238  {
239  ROS_ERROR_STREAM("## ERROR UdpReceiverSocketImpl::Receive(): unexpected payloadlength " << Payloadlength << " byte incl CRC received");
240  break;
241  }
242  bytes_received += chunk_bytes_received;
243  }
244  else if (bytes_received > 0) // continue receiving a msgpack
245  {
246  bytes_received += chunk_bytes_received;
247  }
248  }
249  return (m_running ? bytes_received : 0);
250  }

341  {

366  }
367  #endif

369  sim_servaddr.sin_family = AF_INET;
370  sim_servaddr.sin_port = htons(m_udp_port);
371  bytes_sent = sendto(m_udp_socket, (const char*)message.data(), message.size(), 0, (SOCKADDR*)&sim_servaddr, sizeof(sim_servaddr));
372  if (bytes_sent != message.size())
